  • for(구구단 세로 출력)
    •         #include 

            int main()
            {
                int i = 0; 
                int j = 0; 
                
                //2중 반복문  
                for(i = 1; i <= 9; i++)
                {
                    for(j = 2; j <= 9; j++)
                    {
                        printf("%d * %d = %d\t", j, i, i*j); 		
                    }
                    printf("\n");
                }
                return 0; 
            }
    i = 1; i <= 9; i++
    j = 2; j <= 9; j++
    printf("%d * %d = %d\t", j, i, i*j);
    i 한 번 돌면 j 9번 돌고

    i 하나 오를 때마다 j++ 9까지
    i * j 해서 (j, i, i*j) 출력
    j와 i 순서로 가로 세로 바꾸어짐

        2*1=2 3*1=3 4*1=4 ~~ 
    2*2=4 3*2=6 4*2=8 ~~ 
    2*3=6 3*3=9 4*3=12 ~~~ 
    ~~~

    •         #include <stdio.h> 

            int main()
            {
                //옆으로 갈 때 4즐이고 그 다음 3줄여야 해서 -1 시키기  
                
                int i, j, k=1, f, input, input2; 
                
                printf("숫자 입력: ");
                scanf("%d", &input); 
                int diff = 0; 
               
               for(i=1; i<= input; i++) //행 줄  
                {	  
                   k=i; //k에 행 수를 넣어요  
                   diff = input - 1; //-1 여기서 시킴 
                   
                    for(j=1; j<=i; j++)//열 칸
                    {	 			
                        printf("%3d", k); 
                        k = k + diff; //k 값에 diff 
                        diff = diff - 1; //다음 더 낮아져야 해서 -1 더 추가함  
                      }		
                      printf("\n");  		
                          
               }	
               return 0; 
            }

  • 입력값의 약수 찾기(나누어지는 수)(test8)
    •         #include <stdio.h>

            int main()
            {
                int input, i; 
            
            
                for(;;)
                {	 		
                    printf("숫자 입력 "); 			
                    scanf("%d", &input);
                    
                    if(input == 0)
                    {
                        printf("0이면 퇴출");
                        break; 
                    }
            
                    for(i=1; i<=input; i++)
                    {
                        if(input % i == 0)
                        {
                            printf("%d의 약수는 %d\n", input, i); //나누어지는 숫자  
                        }
                    }
                            
                }
                return 0; 
            }

    scanf로 0입력하면 break로 for문 나감
    입력한 수 만큼까지 i++해서
    나누어 떨어질 수 있는 수를(약수) 다 체크함

    i가 늘어날 때마다 input % i == 0
    (%는 나눈 후 나머지 찾는 연산자)
    나눈 나머지 수가 0이면 출력 (8이라면 1,2,4,8정도)
