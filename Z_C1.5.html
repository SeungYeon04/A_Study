  • 7. 모양만들기 (파일 정리 대충해서 넘어감)
    • #include <stdio.h>
#define SIZE 5

int main() //정보처리기사X 정보기사? 4년재 애들이 푸는? 문제라 함  
{
    int data[SIZE][SIZE] = {0}; 
    int k=1;  
    int start = 0, end = SIZE-1;  //SIZE/2 몫만 가지니 2  

    for(int i = 0; i < SIZE; i++) //2에서 2까지니까 하나  
    {

        for(int j = start; j <= end; j++)
        {	
            data[i][j] = k; 
            k++; 	
        }
        
        //삼각형 만들기 
        if(i < SIZE/2)
        {
            start++;
            end--; 
        } 
        else{
            start--;
            end++; 
        }

    }
    
    for(int i=0; i<SIZE; i++)
    {
        for(int j=0; j<SIZE; j++)
        {
            if(data[i][j] != 0){
                printf("%3d", data[i][j]); //열을 돌려서 0.0 1.0 2.0부터 기입?  
            }
            else 
            {
                printf("   ");
            }
        }
        printf("\n"); 
    }

    return 0; 
}
