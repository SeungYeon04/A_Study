void sort(int input, int (*score)[4])
{
    for(int i = 0; i < 2; i++) //합 정렬 
    {
        if(score[i][3] > score[i+1][3])
        { 
            int tmp = score[i][3]; 
            score[i][3] = score[i+1][3]; 
            score[i+1][3] = tmp; 
            
            for(int j = 0; j < 2; j++)
            {
                int tmpj = score[i][j]; 
                score[i][j] = score[i+1][j]; 
                score[i+1][j] = tmpj;
            }
        }
    } 
}

#define SIZE 20

int main() { 
    
    int num[4];
    int data[SIZE][SIZE] = {0,};
    
    FILE *fin = fopen("c:/input.txt","wt++"); 

    printf("숫자 4개 입력: ");
    scanf("%d %d %d %d", &num[0], &num[1], &num[2], &num[3]);
    fprintf(fin, "%d %d %d %d", num[0], num[1], num[2], num[3]);

    if(fin==NULL){ 	
        printf("파일이 존재하지 않습니다");
        return -1; 
    }
    
    fscanf(fin, "%d %d %d %d", &num[0], &num[1], &num[2], &num[3]);	
    
    fclose(fin);
    
    // 전체 data에  1값 삽입 
    for(int i=num[0]; i<=num[2]; i++) {
        for(int j=num[1]; j<=num[3]; j++) {
            data[i][j]=1;
        }
    }

    // 1값 삽입된 전체 출력 
    for(int i=0; i< SIZE; i++) {
        for(int j=0; j<SIZE; j++) {
            printf("%3d", data[i][j]);
        }
        printf("\n");
    }	
    
    // 파일로 저장하기 
    FILE *fout = fopen("c:/output.txt","wt");

    for(int i=0; i<SIZE; i++) {
        for(int j=0; j<SIZE; j++) {
            fprintf(fout, "%3d", data[i][j]);
        }
            fprintf(fout, "\n");
    }

    
    fclose(fout);
    
    return 0;
}
