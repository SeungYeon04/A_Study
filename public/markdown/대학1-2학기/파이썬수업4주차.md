### 파이썬의 인덱스 쓰기 
```
num1 = 50 
num2 = 45.0 
rlt = "결혼" 

# print("내 나이: %3g 당신의 나이: %.2f 결과: %3s" %  (num1, num2, rlt)) # 하나 이상이면 소괄호 

# format은 0 1 2 하는 인덱스 위치로 설정 
print("내 나이: {0:>3} 당신의 나이: {1:^3} 결과: {2}".format(num1, num2, rlt)) 
print(f"내 나이: {0:<3} 당신의 나이: {1:^3} 결과: {2}".format(num1, num2, rlt)) #f붙이면 인덱스 번호만 출력  

# {0}에서 {0:3}은 3여유 {0:^3} 가운데정렬 {0:<3} 왼쪽 >이건 오른쪽 
```

list = [] #인덱스가 존재하지 않습니다.
