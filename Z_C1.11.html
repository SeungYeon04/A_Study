  • 정렬알고리즘 - 오름차순 내림차순
    • #include <stdio.h>

void N(int *data, int size)
{
    for(int i = 0; i < size; i++)
    {
        printf("%3d ", data[i]);
    }
    printf("\n");
}

int main()
{
    int data[] = {52, 45, 99, 12, 88}; //4바이트5개 = 20 
    int size = sizeof(data)/sizeof(int); // 20/4  
    
    //selection sort(선택정렬) - ascSort (오름차순 정렬) 

    printf("\n기본 : "); 
    N(data, size); 
        
    for(int i = 0; i < size; i++)
    {
        for(int j = i+1; j < size; j++)
        {
            if(data[i]>data[j]) { //오름차  
                int tmp = data[i]; 
                data[i] = data[j]; 
                data[j] = tmp; 
            }
        }
    }
        
    printf("\n오름차순 : "); 
    N(data, size); 
        
    for(int i = 0; i < size; i++)
    {
        for(int j = i+1; j < size; j++)
        {
            if(data[i] < data[j]) { //내림차  
                int tmp = data[i]; 
                data[i] = data[j]; 
                data[j] = tmp; 
            }
        }
    }
    
    printf("\n내림차순 : "); 
    N(data, size); 
}

    기본 : 52 45 99 12 88
    오름차순 : 12 45 52 88 99
    내림차순 : 99 88 52 45 12

void N(int *data, int size)
{
    for(int i = 0; i < size; i++)
    {
        printf("%3d ", data[i]);
    }
    printf("\n");
}

void Or(int *data, int size)
{
    for(int i = 0; i < size; i++)
    {
        for(int j = i+1; j < size; j++)
        {
            if(data[i]>data[j]) { //오름차  
                int tmp = data[i]; 
                data[i] = data[j]; 
                data[j] = tmp; 
            }
        }
    }
}

void Nr(int *data, int size)
{
    for(int i = 0; i < size; i++)
    {
        for(int j = i+1; j < size; j++)
        {
            if(data[i] < data[j]) { //내림차  
                int tmp = data[i]; 
                data[i] = data[j]; 
                data[j] = tmp; 
            }
        }
    }
    
}

int main()
{
    int data[5] = {0}; //4바이트5개 = 20 
    int size = sizeof(data)/sizeof(int); // 20/4 = 5
    
    printf("5개의 수를 입력하세요: \n");
    
    for(int i = 0; i < size; i++)
    {
        scanf("%d", &data[i]); 
    } 

    printf("\n\n기본 : "); 
    N(data, size); 
        
        
    printf("\n오름차순 : "); 
    Or(data, size);
    N(data, size); 
        
    printf("\n내림차순 : "); 
    Nr(data, size);
    N(data, size); 
}

    88
    78
    66
    55
    44

//순위구하기 알고리즘 = Rank()  
//숫자 낮으면 1등 높으면 1등 중에서 == 
// 디폴트값: 1 1 1 1 (0등은 없음) 
//숫자를 자기 자신보다 비교하면서 자기보다 크면 1점씩 올림 = 등수  
//45점인데 95점 있으면 1등 > 2등 비교했는데 55점 있으면 2등 > 3등  

void N(int *data, int size)
{
    printf("\n점수: "); 
    for(int i = 0; i < size; i++)
    {
        printf("%2d ", data[i]);
    }
    printf("\n");
}

void RankP(int *data, int size, int *Rank)
{
    for(int i = 0; i < size; i++)
    {
        printf("%2d등 ", Rank[i]);
    }
    printf("\n");
}

void RankAL(int *data, int size, int *Rank)
{
    for(int i = 0; i < size; i++)
    {
        for(int j = 0; j < size; j++)
        {
            if(data[i] < data[j]) { // 내꺼가 작다면?  
            
                Rank[i]++; //수를 더함  
            }
        }
    } 
    printf("\n");
}

int main()
{
    int data[] = {66, 45, 88, 100, 12}; //4바이트5개 = 20 
    int Rank[] = {1, 1, 1, 1, 1}; 
    int size = sizeof(data)/sizeof(int); // 20/4  

    N(data, size); 
        
    RankAL(data, size, Rank); 	
    
    RankP(data, size, Rank); 
}

int search(int input, int *score, int size)
{
    printf("겁색 Key 값: ");
    scanf("%d", &input);
    
    for(int i = 0; i < size; i++)
    {
        if(score[i] == input)
        {
            return i; //값 있으면 위치 리턴  
        }
    }
    
    return -1; //없으면 -1 리턴  
}

int main()
{
    int input; 

    int score[5]= {12, 33, 44, 55, 99}; 
    int size = sizeof(score)/sizeof(int); // 20/4  
    
    int r = search(input, score, size); 
    
    printf("%d", r+1);

}
