int main()
{
    // asdw 로 움직이는 별  
    int i, j, rp =3, cp= 3; //별 위치 50 중 가운데  
    char c; 
    
    while(1)
    {			
        for(i=1; i<=5; i++)
        {
            for(j=1; j<=5; j++)
            {
                if(i == rp && j == cp) //rp y좌표 cp x좌표  
                {
                    printf("*"); 
                }
                else
                {
                    printf("0");
                }			
            }
            printf("\n");
        }
        
        
        c = getch(); 
        
        if(c == 'a' && cp > 1)
        {
            cp=cp-1; 
        }
        else if(c == 'd' && cp < 5) 
        {
            cp=cp+1; 
        }
        else if(c == 'w' && rp > 1)
        {
            rp=rp-1; 
        }
        else if(c == 's' && rp < 5)
        {
            rp=rp+1; 
        }
        system("cls");
    }   
    return 0;
}
