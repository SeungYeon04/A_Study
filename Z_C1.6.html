  • 1. 3대3으로 순서대로 숫자 출력
    • #include <stdio.h>
    #define SIZE 5
    
    int main()
    {
        int data[SIZE*2-1][SIZE] = {0}; 
        int start =SIZE-1, end = SIZE-1; 
        int k=1;
        
        for(int i = 0; i < SIZE; i++)
        {
            for(int j = start; j <= end; j++)
            {	
                data[j][i] = k; //세로로 출력하는 이중배열  
                k++;	
            }	
            start--; 
            end++;		
        }	
        
                
        for(int i = 0; i < SIZE*2-1; i++)
        {
            for(int j = 0; j < SIZE; j++)
            {
                if(data[i][j])
                    printf("%3d", data[i][j]); 
                else
                    printf("   "); 
            } 
            
            printf("\n"); 
    
        }
    } 
    
    //가로로 저장되는 건 [i][j] 세로로 저장되는 건 [j][i]  
    //값은 j i 로 넣어서 세로출력 될테니까 출력은 i j 그대로 해도 됨 
    //4열 7행으로 했다 
    //5 5 일 경우에는 [스타트3 엔드3] 해야 중간부터 넓어짐  
    /*
    결과: 
                    17
                 10 18
               5 11 19
            2  6 12 20
         1  3  7 13 21
            4  8 14 22
               9 15 23
                 16 24
                    25

  • 2차원 배열에 구구단 2 3 4 값 출력
    • //알고리즘 실기 문제 정처기 
#include <stdio.h>
#define SIZE 9

int main()
{
	int gugu[SIZE/3][SIZE] = {0}; 
	
	for(int i =0; i < SIZE/3; i++)
	{		
		for(int j = 0; j < SIZE; j++)
		{
			gugu[i][j] = (i+2)*(j+1); 
		}
	}
	
	for(int i =0; i < SIZE/3; i++)
	{		
		for(int j = 0; j < SIZE; j++)
		{

			printf("%3d ", gugu[i][j]);
		}
		printf("\n"); 
	}
} 

//이 문제는 2~4까지 곱하기 1~9까지 곱 값 모두 출력  
//i = 0 j = 0 은 배열의 0.0 번째 자리다. 위치정보  
//SIZE 선언은 9까지 도니까 0*1~9 1*1~9 2*1~9 3*1~9 니까 
//0.0에 (i+2)*(j+1) 값 추가 (2)*(1)
// SIZE/3까지 돈다는 거 2 3 4단 값 출력할 거라 0 1 2 3 
/*
결과:  
  2   4   6   8  10  12  14  16  18
  3   6   9  12  15  18  21  24  27
  4   8  12  16  20  24  28  32  36
*/
