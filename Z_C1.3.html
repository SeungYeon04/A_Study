  • (문제 7) 도형넓이구하기 1: 삼각형넓이 2: 사각형넓이
    • #include <stdio.h> 

int main()
{
    int input; 
    int Semo, S1, S2, Nemo, N1, N2; 
    
    printf("1: 삼각형 넓이 & 2: 사각형 넓이\n");
    scanf("%d", &input); 
    
    for(int i = 0; i <= 3; i++)
    {
        if(input == 1)
        {
            printf("밑변ㅡ과 높이ㅣ 입력: ");
            scanf("%d %d", &S1, &S2); 
            Semo = S1 * S2 / 2; 
            printf("\n세모 %d %d는 %d입니다\n", S1, S2, Semo); 
            break; 
        }
        else if(input == 2)
        {
            printf("가로ㅡ 세로ㅣ 입력: ");
            scanf("%d %d", &N1, &N2); 
            Nemo = N1 * N2; 
            printf("\n네모 %d %d는 %d입니다\n", N1, N2, Nemo); 
            break; 
        } 
        else 
        {
            printf("잘못된 값 입력\n");
            printf("1: 삼각형 넓이 & 2: 사각형 넓이\n");
            scanf("%d", &input); 
        }
    } 	
    return 0; 
}

//삼각형의 넓이는 밑변과 높이를 곱한 후 2로 나눈 값입니다. 
//따라서 밑변과 높이가 모두 10인 삼각형의 넓이는 
//10×10=50이 됩니다.

//사각형의 넓이는 가로와 세로를 곱한 값입니다. 
//따라서 가로와 세로가 모두 10인 사각형의 넓이는 
//10×10=100이 됩니다.
    삼각형의 넓이는 밑변과 높이를 곱한 후 2로 나눈 값
    사각형의 넓이는 가로와 세로를 곱한 값입니다
    걍 10 10 받으면 10*10하면 되고 삼각형만 반 나누면 됨.

  • (문제 12) 값 5개로 문제
    • #include <stdio.h>

int main() 
{ 
    int a[4] = {0,0,0,0}; 
    int input, hab = 0, max, min; 
    float mid = 0.0f; 
    
    for(int i = 0; i < 4; i++) 
    { 
        printf("숫자 %d 입력: ", i+1); 
        scanf("%d", &input); 
        a[i] += input; 
        printf("%d\n", a[i]); 
        
        hab = hab + a[i]; 
        
        if(a[0]) //미리 처음꺼 받고 후에 비교하며 값 저장 
        {
            min = a[0]; 
            max = a[0]; 
        }
        if(a[i] < min)
        {
            min = a[i]; 
        }
        if(a[i] > max)
        {
            max = a[i]; 
        }
        mid++; //평균: 나온 숫자 갯수 만큼 나눌 예정 
    } 
    
    printf("합계: %d\n", hab); 
    printf("평균: %.2f\n", hab / mid); 
    printf("최대: %d\n", max); 
    printf("최소: %d\n", min); 
}
