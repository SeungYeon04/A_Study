이 시간은 함수의 기초 뼈대를 알 수 있습니다.

  • 1번째 문제(함수 호출, a b 값 직접 대입)
    • #include <stdio.h>

void getSum(int a, int b){ //함수 (코드를 부품으로 만들고 끌어쓰기)  
    int sum = 0; 
    
    for(int p = a; p <= b; p++)
    {
        sum = sum + p; //누적변수 
    }
    printf("sum = %d\n", sum);
}

int main() //시작함수 
{

    getSum(1, 10); //합 함수 호출문 //1하고 10 받아! 정수값 두 개 a = 1 b = 10값 == sum = 55
    getSum(1, 100); //또 호출 가능 == sum = 5050 
    
    return 0;
}
